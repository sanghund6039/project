create table auth_user(
    auth_id varchar2(100) not null,
    auth_pw varchar2(1000) not null,
    primary key(auth_id)
);
COMMENT ON TABLE auth_user IS '관리자테이블' ;       
COMMENT ON COLUMN auth_user.auth_id IS '관리자아이디' ;       
COMMENT ON COLUMN auth_user.auth_pw IS '관리자비밀번호' ;

CREATE TABLE NOTICE_BOARD(
    BOARD_ID NUMBER,
    BOARD_TITLE VARCHAR2(1000),
    BOARD_CONTENT CLOB,
    BOARD_INSERT_TIME DATE,
    BOARD_INSERT_USER VARCHAR2(100),
    BOARD_UPDATE_TIME DATE,
    BOARD_UPDATE_USER VARCHAR2(100),
    OPEN_YN VARCHAR2(2) DEFAULT 'Y',
    TOP_YN VARCHAR2(2),
    CLICK_CNT NUMBER,
    primary key(BOARD_ID)
);

COMMENT ON TABLE NOTICE_BOARD IS '공지사항게시판' ;       
COMMENT ON COLUMN NOTICE_BOARD.BOARD_ID IS '공지사항 아이디';       
COMMENT ON COLUMN NOTICE_BOARD.BOARD_TITLE IS '공지사항 제목'; 
COMMENT ON COLUMN NOTICE_BOARD.BOARD_CONTENT IS '공지사항 내용'; 
COMMENT ON COLUMN NOTICE_BOARD.BOARD_INSERT_TIME IS '공지사항 입력일'; 
COMMENT ON COLUMN NOTICE_BOARD.BOARD_INSERT_USER IS '공지사항 입력자'; 
COMMENT ON COLUMN NOTICE_BOARD.BOARD_UPDATE_TIME IS '공지사항 수정일'; 
COMMENT ON COLUMN NOTICE_BOARD.BOARD_UPDATE_USER IS '공지사항 수정자'; 
COMMENT ON COLUMN NOTICE_BOARD.OPEN_YN IS '공지사항 공개'; 
COMMENT ON COLUMN NOTICE_BOARD.TOP_YN IS '공지사항 중요'; 
COMMENT ON COLUMN NOTICE_BOARD.CLICK_CNT IS '공지사항 조회수'; 

CREATE TABLE FILE_BOARD(
    FILE_ID NUMBER,
    BOARD_ID NUMBER,
    FILE_NM VARCHAR2(1000),
    FILE_URL VARCHAR2(1000),
    FILE_FULL_URL VARCHAR2(1000),
    FILE_INSERT_USER VARCHAR2(100),
    FILE_INSERT_TIME DATE,
    PRIMARY KEY(FILE_ID)
);

COMMENT ON TABLE FILE_BOARD IS '파일테이블' ;       
COMMENT ON COLUMN FILE_BOARD.FILE_ID IS '파일 아이디';       
COMMENT ON COLUMN FILE_BOARD.BOARD_ID IS '공지사항 아이디'; 
COMMENT ON COLUMN FILE_BOARD.FILE_NM IS '파일이름'; 
COMMENT ON COLUMN FILE_BOARD.FILE_URL IS '파일경로'; 
COMMENT ON COLUMN FILE_BOARD.FILE_FULL_URL IS '파일 전체경로'; 
COMMENT ON COLUMN FILE_BOARD.FILE_INSERT_USER IS '파일 입력자'; 
COMMENT ON COLUMN FILE_BOARD.FILE_INSERT_TIME IS '파일 입력시간'; 

CREATE SEQUENCE NOTICE_SEQ  MINVALUE 1 MAXVALUE 99999999999 INCREMENT BY 1 START WITH 1 NOCACHE   NOCYCLE ;
CREATE SEQUENCE FILE_SEQ  MINVALUE 1 MAXVALUE 99999999999 INCREMENT BY 1 START WITH 1 NOCACHE   NOCYCLE ;
